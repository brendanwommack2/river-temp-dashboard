```js
const last7 = data2026.slice(-7);

const last7Avg = d3.mean(last7, d => d.tmp);
const prior7Avg = d3.mean(prior7, d => d.tmp);
const trendDelta = (prior7.length && last7.length) ? last7Avg - prior7Avg : null;
const trendDirection = trendDelta === null ? "flat" : trendDelta > 0.15 ? "rising" : trendDelta < -0.15 ? "falling" : "steady";
const trendArrow = trendDirection === "rising" ? "↑" : trendDirection === "falling" ? "↓" : "→";
const trendColor = trendDirection === "rising" ? "#B03823" : trendDirection === "falling" ? "#537F1C" : "#705C57";
```

```js
// Tag each point with a run id so same-color points aren't bridged across gaps.
// A line segment connecting a point above the threshold to a point below it
// (or vice versa) is only ONE color in Plot — coloring it by the endpoint's
// "above" flag makes part of that segment visually cross the threshold line
// in the wrong color. To fix this, we interpolate the exact date at which the
// series crosses 15.5°C and insert that crossing point as the shared boundary
// between the two runs. The segment before the crossing is then entirely on
// one side of the threshold, and the segment after is entirely on the other.
const data2026Runs = (() => {
  const result = [];
  let runId = 0;
  data2026.forEach((d, i) => {
    const isAbove = d.tmp >= threshold;
    if (i > 0) {
      const prev = data2026[i - 1];
      const prevIsAbove = prev.tmp >= threshold;
      if (isAbove !== prevIsAbove) {
        // Linear interpolation for the date at which tmp === threshold
        const t = (threshold - prev.tmp) / (d.tmp - prev.tmp);
        const crossDate = new Date(prev.date.getTime() + t * (d.date.getTime() - prev.date.getTime()));
        // Close out the previous run exactly at the threshold
        result.push({ date: crossDate, tmp: threshold, runId, above: prevIsAbove });
        runId++;
        // Start the new run at the same point, new color
        result.push({ date: crossDate, tmp: threshold, runId, above: isAbove });
      }
    }
    result.push({ ...d, runId, above: isAbove });
  });
  return result;
})();
```
