const data2026 = parsed
  .filter(d => d.date?.getFullYear() === 2026 && d.tmp !== null)
  .sort((a, b) => a.date - b.date);

const threshold = 15.5;
const latest = data2026[data2026.length - 1];
const seasonHigh = d3.max(data2026, d => d.tmp);
const seasonHighDate = data2026.find(d => d.tmp === seasonHigh)?.date;
const daysAbove = data2026.filter(d => d.tmp >= threshold).length;

```js
// 7-day trend
const last7 = data2026.slice(-7);
const prior7 = data2026.slice(-14, -7);
const last7Avg = d3.mean(last7, d => d.tmp);
const prior7Avg = d3.mean(prior7, d => d.tmp);
const trendDelta = (prior7.length && last7.length) ? last7Avg - prior7Avg : null;
const trendDirection = trendDelta === null ? "flat" : trendDelta > 0.15 ? "rising" : trendDelta < -0.15 ? "falling" : "steady";
const trendArrow = trendDirection === "rising" ? "↑" : trendDirection === "falling" ? "↓" : "→";
const trendColor = trendDirection === "rising" ? "#B03823" : trendDirection === "falling" ? "#537F1C" : "#705C57";
```

```js
// Year-over-year
const data2025 = parsed
  .filter(d => d.date?.getFullYear() === 2025 && d.tmp !== null)
  .sort((a, b) => a.date - b.date);

const doyOf = d => d3.utcDay.count(d3.utcYear(d), d);
const latestDoy = latest ? doyOf(latest.date) : null;
const sameDay2025 = latestDoy !== null
  ? data2025.find(d => doyOf(d.date) === latestDoy)
  : undefined;
const yoyDelta = (sameDay2025 && latest) ? latest.tmp - sameDay2025.tmp : null;

const data2025ToDate = latestDoy !== null
  ? data2025.filter(d => doyOf(d.date) <= latestDoy)
  : data2025;
```

```js
// Tag each point with a run id so same-color points aren't bridged across gaps
const data2026Runs = (() => {
  let runId = 0;
  let prevAbove = null;
  return data2026.map(d => {
    const isAbove = d.tmp >= threshold;
    if (prevAbove !== null && isAbove !== prevAbove) runId++;
    prevAbove = isAbove;
    return { ...d, runId, above: isAbove };
  });
})();
```
